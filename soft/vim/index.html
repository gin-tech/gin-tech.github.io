設定ファイル.vim.rc

"vi互換にしない
set nocompatible
"タイトルをバッファ名にしない
set notitle
"バックスペースで行頭を削除
set backspace=indent,eol,start
"矩形選択で行末の後ろに置ける
set virtualedit=block
"履歴保存数
set history=500
"バックアップなし
set nobackup
"検索で行末から行頭に戻る
set wrapscan
"括弧の対応をハイライト
set showmatch
"ターミナルからペースト許可
set paste
"相対行番号表示
set relativenumber
"行列番号表示
set ruler
"カーソルライン表示
set cursorline
"カーソル列表示
set cursorcolumn
"TAB,EOFなどを可視化する
set list
set listchars=tab:>-,trail:-
"カーソルの回り込み
set whichwrap=b,s,[,],<,>
"クリップボード
set clipboard+=unnamed
"シンタックス
syntax on
"tabのspace置き換え
set expandtab
"tab幅
set tabstop=3
"インデント幅
set shiftwidth=3
"自動インデント
set autoindent
"スマートインデント
set smartindent
"インクリメントサーチ
set incsearch
"ステータスライン
set laststatus=2
"コマンドライン補完
set wildmenu
